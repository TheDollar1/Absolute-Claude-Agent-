        else:
            state.trend = -1
            state.level = trail

    state.flip_bear = (state.trend == -1 and prev_trend == 1)
    
    return state

def calc_ut_bot(
    closes: np.ndarray,
    highs: np.ndarray,
    lows: np.ndarray,
    sensitivity: float = 1.5,
    atr_period: int = 2,
    prev_trail_buy: float = 0.0,
    prev_trail_sell: float = 0.0
) -> UTBotState:
    """
    Hyper-sensitive UT Bot trigger.
    Period 2 ATR + 1.5 multiplier = sniper mode.
    """
    state = UTBotState()
    n = len(closes)
    
    if n < atr_period + 2:
        return state
    
    atr = _atr(highs, lows, closes, atr_period)
    if atr is None or len(atr) == 0:
        return state
    
    current_close = closes[-1]
    prev_close = closes[-2]
    current_atr = atr[-1]
    
    if current_atr <= 0 or np.isnan(current_atr):
        return state
    
    n_loss = sensitivity * current_atr
    
    # Buy trail (ratchets up when price is above)
    if prev_trail_buy == 0:
        trail_buy = current_close - n_loss
    else:
        if current_close > prev_trail_buy and prev_close > prev_trail_buy:
            trail_buy = max(prev_trail_buy, current_close - n_loss)
        elif current_close < prev_trail_buy and prev_close < prev_trail_buy:
            trail_buy = min(prev_trail_buy, current_close + n_loss)
        else:
            trail_buy = current_close - n_loss if current_close > prev_trail_buy else current_close + n_loss
    
    # Sell trail (ratchets down when price is below)
    if prev_trail_sell == 0:
        trail_sell = current_close + n_loss
    else:
        if current_close > prev_trail_sell and prev_close > prev_trail_sell:
            trail_sell = max(prev_trail_sell, current_close - n_loss)
        elif current_close < prev_trail_sell and prev_close < prev_trail_sell:
            trail_sell = min(prev_trail_sell, current_close + n_loss)
        else:
            trail_sell = current_close - n_loss if current_close > prev_trail_sell else current_close + n_loss
    
    state.trail_buy = trail_buy
    state.trail_sell = trail_sell
    
    # Crossover signals
    ema1 = _ema(closes, 1)  # essentially close
    if ema1 is not None and len(ema1) >= 2:
        # Buy: ema crosses above buy trail
        if ema1[-1] > trail_buy and ema1[-2] <= prev_trail_buy and current_close > trail_buy:
            state.buy_signal = True
        # Sell: sell trail crosses above ema
        if trail_sell > ema1[-1] and prev_trail_sell <= ema1[-2] and current_close < trail_sell:
            state.sell_signal = True
    
    return state

    if n < max(rsi_len, vwap_period, fib_len) + 2:
        return state
    
    # RSI
    rsi = _rsi(closes, rsi_len)
    if rsi is not None and len(rsi) > 0:
        state.rsi = rsi[-1]
    
    # EMA change for momentum confirmation
    ema5 = _ema(closes, 5)
    change_ema5 = ema5[-1] - ema5[-2] if ema5 is not None and len(ema5) >= 2 else 0.0
    
    # RSI momentum state
    positive = state.rsi > pmom and change_ema5 > 0
    negative = state.rsi < nmom and change_ema5 < 0
    
    state.bullish = positive
    state.bearish = negative
    
    # VWAP swing direction (simplified - tracks swing highs/lows)
    if require_vwap:
        # Determine last swing direction based on highestbars/lowestbars
        highest_bar = _highestbars(highs[-vwap_period:])
        lowest_bar = _lowestbars(lows[-vwap_period:])
        state.vwap_bull = highest_bar > lowest_bar  # last pivot was a low ? bullish
    else:
        state.vwap_bull = state.bullish  # pass-through if disabled
    
    # Fib trend (double-smoothed EMA direction)
    if require_fib:
        basis = _ema(_ema_raw(closes, fib_len), fib_len)
        if basis is not None and len(basis) >= 2:
            if basis[-1] > basis[-2]:
                state.fib_trend = 1
            elif basis[-1] < basis[-2]:
                state.fib_trend = -1
    else:
        state.fib_trend = 1 if state.bullish else -1 if state.bearish else 0
    
    return state

def evaluate_matrix(

    ut_buy: bool,
    ut_sell: bool,

    liq_flip_bull: bool,

        else GateResult.FAIL if use_trail_gate
        else GateResult.DISABLED

        mtf_ok_long = True
        mtf_ok_short = True

        mtf_ok_short = m15_trend == -1

        mtf_ok_short = ltf_trend == -1
    
    gates.mtf_gate = (
        GateResult.PASS if (mtf_ok_long or mtf_ok_short)
        else GateResult.FAIL
    )
    
    # -- Regime Gate --
    base_bullish = regime_bullish and (vwap_bull if require_vwap else True)
    base_bearish = regime_bearish and (not vwap_bull if require_vwap else True)
    
    if require_fib:
        regime_bull = base_bullish and fib_trend == 1
        regime_bear = base_bearish and fib_trend == -1
    else:
        regime_bull = base_bullish
        regime_bear = base_bearish
    
    gates.regime_gate = GateResult.PASS if regime_bull or regime_bear else GateResult.FAIL
    
    # -- Combine into Matrix --
    matrix_bull = (regime_bull if enable_regime else True) and mtf_ok_long
    matrix_bear = (regime_bear if enable_regime else True) and mtf_ok_short
    
    # -- Trigger Sources --
    ut_buy_gated = ut_buy and long_trail_ok
    ut_sell_gated = ut_sell and short_trail_ok
    liq_long = liq_flip_bull
    liq_short = liq_flip_bear
    
    # -- Final Signals --
    gates.final_long = (ut_buy_gated or liq_long) and matrix_bull
    gates.final_short = (ut_sell_gated or liq_short) and matrix_bear
    
    return gates

def calc_trade_setup(
    entry: float,
    direction: Direction,
    atr: float,
    swing_low: float,
    swing_high: float,
    ema21: float,
    risk_dollars: float = 25.0,
    sl_buffer_atr: float = 0.5,
    contract_notional: float = 1.0,
    min_lot: float = 0.01,
    trigger: TriggerSource = TriggerSource.UT
) -> TradeSetup:
    """
    Platinum Risk Model.
    Calculates SL, TP1/2/3, position size, and actual risk.
    """
    setup = TradeSetup()
    setup.entry = entry
    setup.direction = direction
    setup.risk_dollars = risk_dollars
    setup.trigger = trigger
    
    if direction == Direction.LONG:
        # SL: structural low, buffered
        sl_base = max(ema21, swing_low) if swing_low > ema21 else swing_low
        sl = sl_base - (atr * sl_buffer_atr)
        if sl >= entry:
            sl = entry - (atr * 1.5)
        risk_dist = entry - sl
    else:
        # SL: structural high, buffered
        sl_base = min(ema21, swing_high) if swing_high < ema21 else swing_high
        sl = sl_base + (atr * sl_buffer_atr)
        if sl <= entry:
            sl = entry + (atr * 1.5)
        risk_dist = sl - entry
    
    setup.sl = sl
    setup.risk_dist = risk_dist
    
    if direction == Direction.LONG:
        setup.tp1 = entry + risk_dist * 1.0
        setup.tp2 = entry + risk_dist * 1.5
        setup.tp3 = entry + risk_dist * 2.0
    else:
        setup.tp1 = entry - risk_dist * 1.0
        setup.tp2 = entry - risk_dist * 1.5
        setup.tp3 = entry - risk_dist * 2.0
    
    # Position sizing
    if risk_dist > 0 and contract_notional > 0:
        setup.size = round(risk_dollars / (risk_dist * contract_notional), 4)
    
    # Trade ID
    now = datetime.now(timezone.utc)
    dir_str = "BUY" if direction == Direction.LONG else "SELL"
    setup.trade_id = f"ATM-{now.strftime('%Y%m%d-%H%M')}-{dir_str}-{trigger.value}"
    
    return setup

def update_memory(memory: AgentMemory, result: TradeResult) -> AgentMemory:
    """
    Updates the agent's memory after each closed trade.
    Adapts confidence_modifier based on recent performance.
    """
    memory.trades.append(result)
    memory.daily_pips += result.pips
    
    if result.pips > 0:
        memory.daily_wins += 1
        memory.gross_wins += result.pips
        memory.consecutive_losses = 0
    else:
        memory.daily_losses += 1
        memory.gross_losses += abs(result.pips)
        memory.consecutive_losses += 1
    
    # Profit factor from last 10 trades
    recent = memory.trades[-10:]
    if len(recent) >= 5:
        wins = sum(t.pips for t in recent if t.pips > 0)
        losses = sum(abs(t.pips) for t in recent if t.pips < 0)
        pf = wins / losses if losses > 0 else 999.0
        memory.last_10_pf.append(pf)
    
    # Adaptive confidence modifier
    if len(memory.last_10_pf) > 0:
        avg_pf = np.mean(memory.last_10_pf[-5:])
        if avg_pf < 1.0:
            memory.confidence_modifier = 0.5   # Reduce position size
        elif avg_pf < 1.5:
            memory.confidence_modifier = 0.75
        elif avg_pf >= 2.5:
            memory.confidence_modifier = 1.25  # Slight increase allowed
        else:
            memory.confidence_modifier = 1.0
    
    # Consecutive loss circuit breaker
    if memory.consecutive_losses >= 4:
        memory.confidence_modifier = 0.0  # Halt trading
    
    return memory

# LOW-LEVEL INDICATOR HELPERS
# --------------------------------------------------------------

def _ema(data: np.ndarray, period: int) -> Optional[np.ndarray]:
    """Exponential Moving Average"""
    if len(data) < period:
        return None
    result = np.empty_like(data)
    result[:period] = np.mean(data[:period])
    multiplier = 2.0 / (period + 1)
    for i in range(period, len(data)):
        result[i] = data[i] * multiplier + result[i-1] * (1 - multiplier)
    return result

def _ema_raw(data: np.ndarray, period: int) -> np.ndarray:
    """EMA returning same-length array (for chaining)"""
    result = _ema(data, period)
    return result if result is not None else data

def _atr(highs: np.ndarray, lows: np.ndarray, closes: np.ndarray, period: int) -> Optional[np.ndarray]:
    """Average True Range"""
    n = len(closes)
    if n < 2:
        return None
    tr = np.empty(n)
    tr[0] = highs[0] - lows[0]
    for i in range(1, n):
        tr[i] = max(highs[i] - lows[i], abs(highs[i] - closes[i-1]), abs(lows[i] - closes[i-1]))
    # Use RMA (Wilder's smoothing) for ATR
    result = np.empty(n)
    result[:period] = np.mean(tr[:period]) if n >= period else tr[0]
    alpha = 1.0 / period
    for i in range(period, n):
        result[i] = alpha * tr[i] + (1 - alpha) * result[i-1]
    return result

def _rsi(closes: np.ndarray, period: int) -> Optional[np.ndarray]:
    """Relative Strength Index"""
    n = len(closes)
    if n < period + 1:
        return None
    deltas = np.diff(closes)
    gains = np.where(deltas > 0, deltas, 0.0)
    losses = np.where(deltas < 0, -deltas, 0.0)
    
    avg_gain = np.mean(gains[:period])
    avg_loss = np.mean(losses[:period])
    
    result = np.empty(n)
    result[:period+1] = 50.0
    
    for i in range(period, n-1):
        avg_gain = (avg_gain * (period - 1) + gains[i]) / period
        avg_loss = (avg_loss * (period - 1) + losses[i]) / period
        rs = avg_gain / avg_loss if avg_loss > 0 else 100.0
        result[i+1] = 100.0 - (100.0 / (1.0 + rs))
    
    return result

def _highestbars(data: np.ndarray) -> int:
    """Returns bars since highest high (negative = past)"""
    if len(data) == 0:
        return 0
    return int(np.argmax(data) - len(data) + 1)

def _lowestbars(data: np.ndarray) -> int:
    """Returns bars since lowest low (negative = past)"""
    if len(data) == 0:
        return 0
    return int(np.argmin(data) - len(data) + 1)
